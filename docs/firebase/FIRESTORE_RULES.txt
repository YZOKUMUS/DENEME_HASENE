rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Kullanıcı istatistikleri - userId ile erişim
    match /user_stats/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Günlük görevler - userId ile erişim
    match /daily_tasks/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Kelime istatistikleri - userId_wordId formatında
    match /word_stats/{docId} {
      allow read, write: if request.auth != null && docId.startsWith(request.auth.uid + '_');
    }
    
    // Favori kelimeler - userId_wordId formatında
    match /favorites/{docId} {
      allow read, write: if request.auth != null && docId.startsWith(request.auth.uid + '_');
    }
    
    // Başarımlar - userId_achievementId formatında
    match /achievements/{docId} {
      allow read, write: if request.auth != null && docId.startsWith(request.auth.uid + '_');
    }
    
    // Rozetler - userId_badgeId formatında
    match /badges/{docId} {
      allow read, write: if request.auth != null && docId.startsWith(request.auth.uid + '_');
    }
  }
}
