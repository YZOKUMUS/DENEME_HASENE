rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Kullanıcı profilleri - anonymous kullanıcılar da erişebilir
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Kullanıcı istatistikleri - anonymous kullanıcılar da erişebilir
    match /user_stats/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Günlük görevler - anonymous kullanıcılar da erişebilir
    match /daily_tasks/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Kelime istatistikleri - anonymous kullanıcılar da erişebilir
    // Hem docId hem de user_id field'ına göre kontrol et
    match /word_stats/{docId} {
      allow read, write: if request.auth != null && 
        (docId.startsWith(request.auth.uid + '_') || 
         resource.data.user_id == request.auth.uid ||
         request.resource.data.user_id == request.auth.uid);
    }
    
    // Favoriler - anonymous kullanıcılar da erişebilir
    match /favorites/{docId} {
      allow read, write: if request.auth != null && 
        (docId.startsWith(request.auth.uid + '_') || 
         resource.data.user_id == request.auth.uid ||
         request.resource.data.user_id == request.auth.uid);
    }
    
    // Başarımlar - anonymous kullanıcılar da erişebilir
    match /achievements/{docId} {
      allow read, write: if request.auth != null && 
        (docId.startsWith(request.auth.uid + '_') || 
         resource.data.user_id == request.auth.uid ||
         request.resource.data.user_id == request.auth.uid);
    }
    
    // Rozetler - anonymous kullanıcılar da erişebilir
    match /badges/{docId} {
      allow read, write: if request.auth != null && 
        (docId.startsWith(request.auth.uid + '_') || 
         resource.data.user_id == request.auth.uid ||
         request.resource.data.user_id == request.auth.uid);
    }
  }
}
